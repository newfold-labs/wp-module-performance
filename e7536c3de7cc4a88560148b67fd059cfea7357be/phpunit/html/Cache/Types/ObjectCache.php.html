498            'host',
499            'port',

501            'password',

593    /**
594     * Parse a scalar constant value from raw wp-config text.
595     *
596     * @param string $raw Raw value from WPConfigTransformer::get_value().
597     * @return mixed|null
598     */
599    private static function parse_wp_config_scalar( $raw ) {
600        $raw = trim( (string) $raw );
601        if ( '' === $raw ) {
602            return null;
603        }
604
605        if ( preg_match( '/^\"(.*)\"$/s', $raw, $m ) ) {
606            return stripcslashes( $m[1] );
607        }
608
609        if ( preg_match( '/^\'(.*)\'$/s', $raw, $m ) ) {
610            return str_replace( array( '\\\\', '\\\'' ), array( '\\', '\'' ), $m[1] );
611        }
612
613        if ( 'true' === strtolower( $raw ) ) {
614            return true;
615        }
616        if ( 'false' === strtolower( $raw ) ) {
617            return false;
618        }
619
620        if ( ctype_digit( $raw ) ) {
621            return (int) $raw;
622        }
623
624        if ( is_numeric( $raw ) ) {
625            return 0 + $raw;
626        }
627
628        return null;
629    }
630
631    /**
632     * Parse Redis 6 ACL-style password from wp-config when the value is not a scalar string.
633     *
634     * Supports short and long PHP array syntax with two string elements (username and password).
635     *
636     * @param string $raw Raw value from WPConfigTransformer::get_value().
637     * @return array{0:string,1:string}|null
638     */
639    private static function parse_wp_config_redis_acl_password( $raw ) {
640        $raw = trim( (string) $raw );
641        if ( '' === $raw ) {
642            return null;
643        }
644
645        // Short array syntax with two quoted elements.
646        if ( '' !== $raw && '[' === $raw[0] ) {
647            if ( preg_match( '/^\[\s*\'([^\']*)\'\s*,\s*\'([^\']*)\'\s*\]$/s', $raw, $m ) ) {
648                return array( $m[1], $m[2] );
649            }
650            if ( preg_match( '/^\[\s*"([^"\\\\]*(?:\\\\.[^"\\\\]*)*)"\s*,\s*"([^"\\\\]*(?:\\\\.[^"\\\\]*)*)"\s*\]$/s', $raw, $m ) ) {
651                return array( stripcslashes( $m[1] ), stripcslashes( $m[2] ) );
652            }
653        }
654
655        // Long array syntax with two single-quoted elements.
656        if ( preg_match( '/^array\s*\(\s*\'([^\']*)\'\s*,\s*\'([^\']*)\'\s*\)\s*$/is', $raw, $m ) ) {
657            return array( $m[1], $m[2] );
658        }
659        if ( preg_match( '/^array\s*\(\s*"([^"\\\\]*(?:\\\\.[^"\\\\]*)*)"\s*,\s*"([^"\\\\]*(?:\\\\.[^"\\\\]*)*)"\s*\)\s*$/is', $raw, $m ) ) {
660            return array( stripcslashes( $m[1] ), stripcslashes( $m[2] ) );
661        }
662
663        return null;
664    }
